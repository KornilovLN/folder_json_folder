# YAML-файл содержит конфигурацию для извлечения разл. файлов из архива.
#   В данном случае, файлы классифицируются по их пути и расширению и 
#   направляются в соответствующие целевые директории.


# encoding: 'koi8-r'  Уст. кодировку для извлеченных данных на 'koi8-r'

# targets:            Опред. цели и соотв. шаблоны файлов для извлечения
# 'listinfo.cks': ''  файл 'listinfo.cks' извлечь в корневую директорию
# 'info.ver': ''      файл 'info.ver' извлечь в корневую директорию

## database
# 'RPMS/info_database.+\.rpm':  Опред. шаблон для RPM пакетов баз данных
#   '.+\.gbk': 'db'             Извл. файлы с расш. .gbk в дир. 'db'

## exchange lists
# 'RPMS/info_rex-.+\.i386.rpm': Опред. шаблон RPM пакетов списков обмена
#   '.+/GKT/[^/]+\.000': 'db/ex'  Извл. файлы в поддир. 'GKT'
#                                 с расшир. .000 в директорию 'db/ex'

## mnemos
# 'RPMS/info_ris-.+\.i386.rpm':   Опред. шаблон для RPM пакетов mnemos
#   '.+/ris/grset': 'ris/etc'     Извл. файл 'grset' из'ris 'в 'ris/etc'
#   '.+/ris/[^/]+\.xml': 'ris'    Извл. файлы .xml из'ris' в 'ris'
#   '.+/ris/[^/]+\.(bmp|png)': 'ris'  
#                                 Извл. файлы .bmp или .png
#                                 из 'ris' в директорию 'ris'
#   '.+/ris/kfb/[^/]+\.xml': 'ris/kfb'  
#                                 файлы .xml из 'ris/kfb' в 'ris/kfb'
#   '.+/ris/kfb/[^/]+\.(bmp|png)': 'ris/kfb'
#                                 файлы .bmp или .png из 'ris/kfb'
#                                 в директорию 'ris/kfb'
#   '.+/ris/LDG$|.+/ris/LDS$|.+/ris/LSG$|.+/ris/ALGOR$': 'ris'  
#                                 Извл. файлы LDG, LDS, LSG или ALGOR
#                                 из 'ris' в директорию 'ris'

## Обработка баз данных:
# 'RPMS/info_database.+\.rpm': 
#     Определяет файлы с именами, соответствующими этому шаблону
#     (используется регулярное выражение).
#     Эти файлы находятся в директории RPMS и
#     содержат слово info_database в своем имени.
# '.+\.gbk': 'db':
#     Указывает, что все файлы с расширением .gbk из этих архивов
#     будут извлечены в директорию db.

## Обработка списков обмена:
# 'RPMS/info_rex-.+\.i386.rpm':
#     Определяет файлы с именами, соответствующими этому шаблону.
#     Эти файлы находятся в директории RPMS и
#     содержат слово info_rex- в своем имени.
# '.+/GKT/[^/]+\.000': 'db/ex':
#     Указывает, что все файлы с именами, заканчивающимися на .000
#     в поддиректории GKT, будут извлечены в директорию db/ex.

## Обработка мнемонических данных:

# 'RPMS/info_ris-.+\.i386.rpm':
#     Определяет файлы с именами, соответствующими этому шаблону.
#     Эти файлы находятся в директории RPMS и
#     содержат слово info_ris- в своем имени.
# '.+/ris/grset': 'ris/etc':
#     Указывает, что файлы с именами, содержащими grset
#     в поддиректории ris, будут извлечены в директорию ris/etc.
# '.+/ris/[^/]+\.xml': 'ris':
#     Указывает, что все XML-файлы в поддиректории ris
#     будут извлечены в директорию ris.
# '.+/ris/[^/]+\.(bmp|png)': 'ris':
#     Указывает, что все файлы с расширениями .bmp и .png
#     в поддиректории ris будут извлечены в директорию ris.
# '.+/ris/kfb/[^/]+\.xml': 'ris/kfb':
#     Указывает, что все XML-файлы в поддиректории ris/kfb
#     будут извлечены в директорию ris/kfb.
# '.+/ris/kfb/[^/]+\.(bmp|png)': 'ris/kfb':
#     Указывает, что все файлы с расширениями .bmp и .png
#     в поддиректории ris/kfb будут извлечены в директорию ris/kfb.
# '.+/ris/LDG$|.+/ris/LDS$|.+/ris/LSG$|.+/ris/ALGOR$': 'ris':
#     Указывает, что файлы с именами,
#     заканчивающимися на LDG, LDS, LSG или ALGOR в поддиректории ris,
#     будут извлечены в директорию ris.

## Объяснение процесса
#     Шаблоны регулярных выражений используются для идентификации файлов
#     внутри архивов. Эти файлы могут находиться в разных поддиректориях
#     и иметь различные расширения.
#
#     Целевые пути указывают, в какие директории будут извлечены файлы
#     после их идентификации. Это позволяет структурировать извлеченные
#     данные в соответствии с их типами и назначением.

# Исходный файл без комментариев:
encoding: 'koi8-r'
targets:
  'listinfo.cks': ''
  'info.ver': ''

  ## database
  'RPMS/info_database.+\.rpm':
      '.+\.gbk': 'db'

  ## exchange lists
  'RPMS/info_rex-.+\.i386.rpm':
      '.+/GKT/[^/]+\.000': 'db/ex'

  ## mnemos
  'RPMS/info_ris-.+\.i386.rpm':
      '.+/ris/grset': 'ris/etc'
      '.+/ris/[^/]+\.xml': 'ris'
      '.+/ris/[^/]+\.(bmp|png)': 'ris'
      '.+/ris/kfb/[^/]+\.xml': 'ris/kfb'
      '.+/ris/kfb/[^/]+\.(bmp|png)': 'ris/kfb'
      '.+/ris/LDG$|.+/ris/LDS$|.+/ris/LSG$|.+/ris/ALGOR$': 'ris'
