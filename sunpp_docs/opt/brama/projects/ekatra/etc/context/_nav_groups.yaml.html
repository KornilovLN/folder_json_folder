_nav_groups.yaml

# Этот YAML-файл определяет структуру меню для раздела "Группы".
# Корневой элемент меню использует иконку fa fa-tags и имеет
# заголовок "Группы".
# Разрешение доступа установлено на permit-all, что означает,
# что этот раздел меню доступен для всех пользователей.
# В разделе children определен один дочерний элемент меню 
# для "Приватных групп". Для него указан URL-адрес (href),
# идентификатор (id) и заголовок (title).
# Этот файл используется для построения навигационного меню
#в пользовательском интерфейсе приложения, в частности, для
#отображения раздела "Группы" и подраздела "Приватные группы".

# Определение класса иконки для элемента меню
cssClass: fa fa-tags

# Разрешение доступа для всех
permit: permit-all

# Заголовок элемента меню
title: Группы

# Дочерние элементы меню
children:

  # Элемент меню для приватных групп пользователей
  - href: ../user/groups  # URL-адрес
    id: user/groups  # Идентификатор
    title: Приватные группы  # Заголовок
