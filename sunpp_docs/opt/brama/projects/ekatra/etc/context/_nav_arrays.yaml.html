# Этот YAML-файл определяет структуру меню для раздела "Массивы".
# Корневой элемент меню имеет идентификатор arr,
# заголовок "Массивы" и использует иконку fa fa-server.
# В разделе children определены дочерние элементы меню.
# В данный момент присутствует только один активный дочерний
# элемент для "УВС энергоблока №3".
# Для него указан URL-адрес (href), идентификатор (id),
# файл с дополнительной структурой меню (menu) 
# и заголовок (title).
# Второй дочерний элемент для "ПАМС" закомментирован.
# Этот файл используется для построения навигационного меню 
# в пользовательском интерфейсе приложения.

# Определение элемента меню "Массивы"
id: arr
title: Массивы
cssClass: fa fa-server  # Класс иконки для элемента меню
permit: permit-all  # Разрешение доступа для всех

# Дочерние элементы меню
children:
  # Элемент меню для УВС энергоблока №3
  - href: u3/arr  # URL-адрес
    id: arr/u3  # Идентификатор
    menu: u3/arr/nav.json  # Файл с дополнительной структурой меню
    title: УВС энергоблока №3
