# Определение единиц измерения и их преобразований

'1/мин/c': 'об/мин/c'  # Преобразование единицы измерения "1/мин/с" в "об/мин/с"

'мкСм/см': 'мкСм/см'  # Преобразование единицы измерения "мкСм/см" в "мкСм/см"

'% об.': 'проц.об'  # Преобразование единицы измерения "% об." в "проц.об"

'(кгС/см²)/ч': 'кгс/см2/ч'  # Преобразование единицы измерения "(кгС/см²)/ч" в "кгс/см2/ч"

'гПа':
 - 'HPa'  # Преобразование единицы измерения "гПа" в "HPa"
 - 'ГПА'  # Преобразование единицы измерения "гПа" в "ГПА"
