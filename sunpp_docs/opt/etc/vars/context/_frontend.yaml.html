# Назначение файла и описание:
# Этот файл содержит конфигурацию для фронтенд-сервера Nginx.
# Он определяет директории для хранения данных и настройки виртуальных хостов.

# Директории для хранения данных Nginx
nginx_data_dirs:
  - /var/ekatra/install
  - /var/ekatra/spd

# Конфигурация виртуальных хостов Nginx
nginx_virtual_hosts:
  spd:
    locations:
      # Локация для /spd
      - path: '^~ /spd'
        root: /var/ekatra
        
      # Локация для /install
      - path: /install
        root: /var/ekatra
        
      # Локация для корневого пути
      - path: /
        root: /var/www
        try_files: $uri $uri/ =404
        
      # Локация для статических файлов
      - path: '~* \.(?:css|js|jpg|jpeg|gif|png|ico|svg|svgz)
        root: /var/www
        try_files: $uri $uri/ =404
