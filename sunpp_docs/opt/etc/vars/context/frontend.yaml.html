nginx_data_dirs:
  - /var/ekatra/install
  - /var/ekatra/spd

nginx_virtual_hosts:
  spd:
    locations:
      - path: '^~ /spd'
        root: /var/ekatra
        
      - path: /install
        root: /var/ekatra
        
      - path: /
        root: /var/www
        try_files: $uri $uri/ =404
        
      - path: '~* \.(?:css|js|jpg|jpeg|gif|png|ico|svg|svgz)$'
        root: /var/www
        try_files: $uri $uri/ =404
