main.yaml

- name: get bridge services items
  find:
    paths: /var/ekatra/tmp/service/bridge
    patterns: '*.yaml'
  register: bridge_services

- name: get collect services items
  find:
    paths: /var/ekatra/tmp/service/collect
    patterns: '*.yaml'
  register: collect_services

- name: Ensure local dirs
  file: path=/var/ekatra/compose/ state=directory mode=0777

- name: make compose config for system services
  template:
    src: '{{ ekatra_mode }}-compose.yaml'
    dest: "/var/ekatra/compose/{{ ekatra_domain }}{{ ekatra_mode }}.yaml"
