#!/usr/bin/env bash
# print sorted massive name;index in massive;nameof algor;nameof img

xmlstarlet sel -T -t -m //param -v @mas -o ';' -v @nom -o ';' \
  -m 'ancestor::params' -v @algor -o ';' -m 'parent::*/img' -v @name -o ';' -v @format -n $1/*.xml \
   | sort --field-separator=';' -V| uniq
