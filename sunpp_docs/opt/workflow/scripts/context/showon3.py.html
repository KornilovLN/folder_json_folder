#! /usr/bin/env python
import sys
import json
from pathlib import Path
from ekatra.compiler import fixnames


def main():
    for line in sys.stdin:
        try:
            line = line[:-1].split(';')
            if len(line) >= 3:
                a = line[0]
                i = int(line[1])
                try:
                    iden = line[2]
                except:
                    iden = None
                f = Path(line[3]).stem
                print(json.dumps(dict(FID='f'+fixnames.fix_name(f), SHIFR=u'{}{:03d}'.format(a, i), IDEN=iden), ensure_ascii=False))
        except:
            continue


if __name__ == '__main__':
    main()
